### Law Proposal Event
country_event = {
	id = sc_legislation.1
	title = sc_legislation.1.name
	desc = sc_legislation.1.desc
	picture = GFX_evt_galactic_senate

	is_triggered_only = yes

	immediate = {
		every_country = {
			limit = {
				OR = {
					is_in_federation_with = ROOT
					is_country = ROOT
				}
			}
			set_timed_country_flag = { flag = sc_proposal_ongoing days = 14 } #changed so that if you close window you're not locked out forever
		}
	}
	
	option = { #Level 1 - Autonomous Laws
		name = sc_legislation.1.a
		custom_tooltip = sc_legislation.1.a.tooltip
		hidden_effect = {
			country_event = {
				id = sc_legislation.10
			}
		}
	}
	option = { #Level 2 - Federal Laws
		name = sc_legislation.1.b
		custom_tooltip = sc_legislation.1.b.tooltip
		trigger = {
			OR = {
				has_policy_flag = sc_federal_federal_laws
				has_policy_flag = sc_federal_unified_laws
			}
		}
		hidden_effect = {
			country_event = {
				id = sc_legislation.20
			}
		}
	}
	option = { #Level 3 - Unified Laws
		name = sc_legislation.1.c
		custom_tooltip = sc_legislation.1.c.tooltip
		trigger = {
			has_policy_flag = sc_federal_unified_laws
		}
		hidden_effect = {
			country_event = {
				id = sc_legislation.30
			}
		}
	}
	option = { #Change Mind
		name = sc_legislation.exit
		custom_tooltip = sc_legislation.exit.tooltip
	}
}
